# Hybrid search combining dense and sparse retrieval
async def hybrid_search(query: str, top_k: int = 10):
    # Dense vector search
    dense_results = await qdrant.search(
        collection_name="documents",
        query_vector=embed(query),
        limit=top_k * 2
    )
    
    # Sparse BM25 search
    sparse_results = bm25_index.search(query, top_k * 2)
    
    # Fusion with weighted scores
    combined = reciprocal_rank_fusion(
        dense_results, 
        sparse_results,
        weights=[0.7, 0.3]
    )
    
    # Cross-encoder reranking
    return rerank_with_cross_encoder(combined, query, top_k)

# Before: O(nÂ²) complexity
def find_duplicates_naive(papers):
    duplicates = []
    for i, paper1 in enumerate(papers):
        for paper2 in papers[i+1:]:
            if similarity(paper1, paper2) > 0.95:
                duplicates.append((paper1, paper2))
    return duplicates

# After: O(n log n) with LSH
def find_duplicates_optimized(papers):
    lsh = MinHashLSH(threshold=0.95)
    for idx, paper in enumerate(papers):
        lsh.insert(f"doc_{idx}", get_minhash(paper))
    
    return [(papers[i], papers[j]) 
            for i, neighbors in enumerate(lsh.query_all())
            for j in neighbors if i < j]

Reduced duplicate detection time from hours to minutes using LSH
